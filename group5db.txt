-- Create database
CREATE DATABASE group5db;

-- Create posts table
CREATE TABLE posts (
    pid SERIAL NOT NULL,
    caption VARCHAR(100) NOT NULL,
    image_path TEXT DEFAULT('images/empty.jpeg') NOT NULL,
    post_date VARCHAR(32) NOT NULL,
    PRIMARY KEY(pid)
);

-- CREATE comments table
CREATE TABLE comments (
    cid SERIAL NOT NULL,
    pid INT NOT NULL,
    name VARCHAR(50) NOT NULL,
    text VARCHAR(500) NOT NULL,
    comment_date VARCHAR(32) NOT NULL,
    PRIMARY KEY(cid),
    FOREIGN KEY(pid) REFERENCES posts(pid)
);
